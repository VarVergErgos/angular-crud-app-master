2    import { Injectable } from '@angular/core';
3    import { Observable } from 'rxjs';
4    
5    @Injectable({
6      providedIn: 'root',

11     addEmployee(data: any): Observable<any> {

13     }
14   
15     updateEmployee(id: number, data: any): Observable<any> {
16       return this._http.put(`http://localhost:3000/employees/${id}`, data);
17     }
18

20       return this._http.get('http://localhost:3000/employees');
21     }
22   
23     deleteEmployee(id: number): Observable<any> {
